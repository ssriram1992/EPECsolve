758 arma::sp_mat TranspCost(unsigned int n) {
759  arma::sp_mat TrCo(n, n);
760  for (unsigned int i = 0; i < n; ++i) {
761  for (unsigned int j = i; j < n; ++j) {
762  TrCo(i, j) = j - i;
763  TrCo(j, i) = j - i;
764  }
765  }
766  return TrCo;
767 }
