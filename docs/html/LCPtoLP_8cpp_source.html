330  unsigned int nPoly{static_cast<unsigned int>(Ai->size())};
331  unsigned int nC{static_cast<unsigned int>(Ai->front()->n_cols)};
332 
333  // Count the number of variables in the convex hull.
334  unsigned int nFinCons{0}, nFinVar{0};
335  // Error check
336  if (nPoly == 0) throw "Empty vector of polyhedra given!"; // There should be at least 1 polyhedron to consider
337  if (nPoly != bi->size()) throw "Inconsistent number of LHS and RHS for polyhedra";
338  for(unsigned int i=0; i!=nPoly; i++)
339  {
340  if (Ai->at(i)->n_cols != nC) throw "Inconsistent number of variables in the polyhedra " + to_string(i) + "; " + to_string(Ai->at(i)->n_cols)+"!="+to_string(nC);

342  nFinCons += Ai->at(i)->n_rows;
343  }
344  unsigned int FirstCons = nFinCons;
345  if(Acom.n_rows >0 &&Acom.n_cols != nC) throw "Inconsistent number of variables in the common polyhedron";
346  if(Acom.n_rows >0 &&Acom.n_rows != bcom.n_rows) throw "Inconsistent number of rows in LHS and RHS in the common polyhedron";
347 
348  // 2nd constraint in Eqn 4.31 of Conforti - twice so we have 2 ineq instead of 1 eq constr
349  nFinCons += nC*2;
350  // 3rd constr in Eqn 4.31. Again as two ineq constr.
351  nFinCons += 2;
352  // Common constraints
353  nFinCons += Acom.n_rows;
354 
355  nFinVar = nPoly*nC + nPoly + nC; // All x^i variables + delta variables+ original x variables
356  A.resize(nFinCons, nFinVar); b.resize(nFinCons);
357  A.zeros(); b.zeros();
358  // Counting rows completed
359  unsigned int complRow{0};
360  if (VERBOSE){cout<<"In Convex Hull computation!"<<endl;}
361  for(unsigned int i = 0; i<nPoly; i++)
362  {
363  unsigned int nConsInPoly = complRow+Ai->at(i)->n_rows ;
364  // First constraint in (4.31)
365  A.submat(complRow, i*nC, complRow+nConsInPoly-1, (i+1)*nC-1) = *Ai->at(i);
366  // First constraint RHS
367  A.submat(complRow, nPoly*nC+i, complRow+nConsInPoly-1, nPoly*nC+i) = -*bi->at(i);
368  // Second constraint in (4.31)
369  for(unsigned int j=0; j<nC;j++)
370  {
371  A.at(FirstCons+2*j,(i*nC)+j) = 1;
372  A.at(FirstCons+2*j+1,(i*nC)+j) = -1;
373  }
374  // Third constraint in (4.31)
375  A.at(FirstCons + nC*2, nPoly*nC + i) = 1;
376  A.at(FirstCons + nC*2 + 1, nPoly*nC + i) = -1;
377  }
378  // Second Constraint RHS
379  for(unsigned int j=0; j<nC;j++)
380  A.at(FirstCons+2*j, nPoly*nC + nPoly +j) = -1;
381  // Third Constraint RHS
382  b.at(FirstCons + nC*2) = 1;
383  b.at(FirstCons + nC*2+1) = -1;
384  // Common Constraints
385  if(Acom.n_rows>0)
386  {
387  b.subvec(FirstCons+2*nC+2, nFinCons-1) = bcom;
388  A.submat(FirstCons+2*nC+2, nPoly*nC+nPoly,
389  nFinCons-1, nFinVar-1) = Acom;
390  }
391  return 0;
392 }

665  }

694 Game::LCP::FixToPoly(const vector<short int> *Fix,
695  bool checkFeas,
696  bool custom,
697  vector<arma::sp_mat*> *custAi,
698  vector<arma::vec*> *custbi
699  )
708 {
709  arma::sp_mat *Aii = new arma::sp_mat(nR, nC);
710  arma::vec *bii = new arma::vec(nR, arma::fill::zeros);
711  for(unsigned int i=0;i<this->nR;i++)
712  {
713  if(Fix->at(i) == 0) throw "Error in FixToPoly";
714  if(Fix->at(i)==1) // Equation to be fixed top zero
715  {
716  for(auto j=this->M.begin_row(i); j!=this->M.end_row(i); ++j)
717  if(!this->isZero((*j))) Aii->at(i, j.col()) = (*j); // Only mess with non-zero elements of a sparse matrix!
718  bii->at(i) = this->q(i);
719  }
720  else // Variable to be fixed to zero, i.e. x(j) <= 0 constraint to be added
721  {
722  unsigned int varpos = (i>this->LeadStart)?i+this->nLeader:i;
723  Aii->at(i, varpos) = 1;
724  bii->at(i) = 0;
725  }
726  }
727  bool add = !checkFeas;
728  if(checkFeas)
729  {
730  unsigned int count{0};
731  try
732  {
733  makeRelaxed();
734  GRBModel* model = new GRBModel(this->RlxdModel);
735  for(auto i:*Fix)
736  {
737  if(i>0) model->getVarByName("z_"+to_string(count)).set(GRB_DoubleAttr_UB,0);
738  if(i<0) model->getVarByName("x_"+to_string(count>this->LeadStart?count+nLeader:i)).set(GRB_DoubleAttr_UB,0);
739  count++;
740  }
741  model->optimize();
742  if(model->get(GRB_IntAttr_Status) == GRB_OPTIMAL) add = true;
743  delete model;
744  }
745  catch(const char* e) { cerr<<"Error in Game::LCP::FixToPoly: "<<e<<endl; throw;}
746  catch(string e) { cerr<<"String: Error in Game::LCP::FixToPoly: "<<e<<endl; throw;}
747  catch(exception &e) { cerr<<"Exception: Error in Game::LCP::FixToPoly: "<<e.what()<<endl; throw;}
748  catch(GRBException &e) {cerr<<"GRBException: Error in Game::LCP::FixToPoly: "<<e.getErrorCode()<<": "<<e.getMessage()<<endl;throw;}
749  }
750  if(add)
751  {
752  custom?this->Ai->push_back(Aii):custAi->push_back(Aii);
753  custom?this->bi->push_back(bii):custbi->push_back(bii);
754  }
755  if(VERBOSE) cout<<"Pushed a new polyhedron! No: "<<Ai->size()<<endl;
756  return *this;
757 }
