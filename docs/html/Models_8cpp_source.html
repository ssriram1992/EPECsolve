180  arma::sp_mat Q(1, 1), C(1, LeadVars + Params.n_followers - 1);
181  // Two constraints. One saying that you should be less than capacity
182  // Another saying that you should be less than leader imposed cap!
183  arma::sp_mat A(1, Loc.at(Models::LeaderVars::End) - 1), B(1, 1);
184  arma::vec c(1), b(1);

192  c.zeros();
193  // Objective
194  Q(0, 0) = Params.FollowerParam.costs_quad.at(follower) + 2 * Params.DemandParam.beta;

200

203

205  //A(1, (Params.n_followers - 1) + 2 + follower) = 0;

208  b(0) = Params.FollowerParam.capacities.at(follower);
