154 void
155 Models::EPEC::make_LL_QP(const LeadAllPar& Params,
156  const unsigned int follower,
157  Game::QP_Param* Foll,
158  const Models::LeadLocs& Loc
159  ) const noexcept
164 {
165  const unsigned int LeadVars = Loc.at(Models::LeaderVars::End) - Params.n_followers;
166  arma::sp_mat Q(1,1), C(1, LeadVars + Params.n_followers - 1);
167  // Two constraints. One saying that you should be less than capacity
168  // Another saying that you should be less than leader imposed cap!
169  arma::sp_mat A(2, Loc.at(Models::LeaderVars::End) - 1), B(2, 1);
170  arma::vec c(1), b(2);
171  c.fill(0); b.fill(0);
172  A.zeros(); B.zeros(); C.zeros(); b.zeros(); Q.zeros(); c.zeros();
173  // Objective
174  Q(0, 0) = Params.FollowerParam.costs_quad.at(follower) + 2*Params.DemandParam.beta;
175  c(0) = Params.FollowerParam.costs_lin.at(follower) - Params.DemandParam.alpha;
176 
177  arma::mat Ctemp(1, Loc.at(Models::LeaderVars::End)-1, arma::fill::zeros);
178  Ctemp.cols(0, Params.n_followers-1).fill(Params.DemandParam.beta); // First n-1 entries and 1 more entry is Beta
179  Ctemp(0, Params.n_followers) = -Params.DemandParam.beta; // For q_exp
180 
181  Ctemp(0, (Params.n_followers-1)+2+Params.n_followers+follower ) = 1; // q_{-i}, then import, export, then tilde q_i, then i-th tax
182 
183  C = Ctemp;
184  // A(1, (Params.n_followers-1)+2 + follower) = -1;
185  B(0,0)=1; B(1,0) = -1;
186  b(0) = Params.FollowerParam.capacities.at(follower);
187  b(1) = - Params.FollowerParam.capacities.at(follower)*0.05;
188 
189  Foll->set(std::move(Q), std::move(C), std::move(A), std::move(B), std::move(c), std::move(b));
190 }
