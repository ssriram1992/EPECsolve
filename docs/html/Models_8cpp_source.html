162  arma::sp_mat Q(1, 1), C(1, LeadVars + Params.n_followers - 1);
163  // Two constraints. One saying that you should be less than capacity
164  // Another saying that you should be less than leader imposed cap!
165  arma::sp_mat A(2, Loc.at(Models::LeaderVars::End) - 1), B(2, 1);
166  arma::vec c(1), b(2);
167  c.fill(0);
168  b.fill(0);
169  A.zeros();
170  B.zeros();
171  C.zeros();
172  b.zeros();
173  Q.zeros();
174  c.zeros();
175  // Objective
176  Q(0, 0) = Params.FollowerParam.costs_quad.at(follower) + 2 * Params.DemandParam.beta;
177  c(0) = Params.FollowerParam.costs_lin.at(follower) - Params.DemandParam.alpha;
178 
179  arma::mat Ctemp(1, Loc.at(Models::LeaderVars::End) - 1, arma::fill::zeros);
180  Ctemp.cols(0, Params.n_followers - 1).fill(Params.DemandParam.beta); // First n-1 entries and 1 more entry is Beta
181  Ctemp(0, Params.n_followers) = -Params.DemandParam.beta; // For q_exp
182 
183  Ctemp(0, (Params.n_followers - 1) + 2 + Params.n_followers +
184  follower) = 1; // q_{-i}, then import, export, then tilde q_i, then i-th tax
185 
186  C = Ctemp;
187  //A(1, (Params.n_followers - 1) + 2 + follower) = 0;
188  //Produce positive (zero) quantities and less than the cap
189  B(0, 0) = 1;
190  B(1, 0) = -1;
191  b(0) = Params.FollowerParam.capacities.at(follower);
192  b(1) = 0; // - Params.FollowerParam.capacities.at(follower)*0.05;
