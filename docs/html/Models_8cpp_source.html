269  Ctemp(0, (Params.n_followers - 1) + 2 + Params.n_followers + 0) =
270  Params.FollowerParam.emission_costs.at(
271  follower); // q_{-i}, then import, export, then tilde q_i, then only
272  // tax var
273 
274  C = Ctemp;
275  // A(1, (Params.n_followers - 1) + 2 + follower) = 0;
276  // Produce positive (zero) quantities and less than the cap
277  B(0, 0) = 1;
278  b(0) = Params.FollowerParam.capacities.at(follower);
279

730  LeadLocs &Loc = this->Locations.at(i);
731  // Counts the number of countries from which the current country imports
732  for (auto val = TranspCosts.begin_col(i); val != TranspCosts.end_col(i);
733  ++val)
734  nImp++;

736  this->nImportMarkets.at(i) = (nImp);

741

743  // total imports
744  arma::vec a(Loc.at(Models::LeaderVars::End) - LL_Nash.getNduals(),
745  arma::fill::zeros);
746  a.at(Loc.at(Models::LeaderVars::NetImport)) = -1;
747  a.subvec(Loc.at(LeaderVars::CountryImport),
748  Loc.at(LeaderVars::CountryImport + 1) - 1)
749  .ones();
750 
751  LL_Nash.addDummy(nImp, Loc.at(Models::LeaderVars::CountryImport));
752  LL_Nash.addLeadCons(a, 0).addLeadCons(-a, 0);
753  } else {
754  Game::NashGame &LL_Nash = *this->countries_LL.at(i).get();

756  // Set imports and exporta to zero

758  arma::fill::zeros);
