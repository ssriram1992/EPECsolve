142  const unsigned int LeadVars = Loc.at(Models::LeaderVars::End) - Params.n_followers;
143  arma::sp_mat Q(1,1), C(1, LeadVars + Params.n_followers - 1);
144  // Two constraints. One saying that you should be less than capacity
145  // Another saying that you should be less than leader imposed cap!
146  arma::sp_mat A(2, Loc.at(Models::LeaderVars::End) - 1), B(2, 1);
147  arma::vec c(1), b(2);

149  A.zeros(); B.zeros(); C.zeros(); b.zeros(); Q.zeros(); c.zeros();
150  // Objective
151  Q(0, 0) = Params.FollowerParam.costs_quad.at(follower) + 2*Params.DemandParam.beta;
152  c(0) = Params.FollowerParam.costs_lin.at(follower) - Params.DemandParam.alpha;
153

155  Ctemp.cols(0, Params.n_followers-1).fill(Params.DemandParam.beta); // First n-1 entries and 1 more entry is Beta
156  Ctemp(0, Params.n_followers) = -Params.DemandParam.beta; // For q_exp
157 
158  Ctemp(0, (Params.n_followers-1)+2+Params.n_followers+follower ) = 1; // q_{-i}, then import, export, then tilde q_i, then i-th tax
159 
160  C = Ctemp;
161  A(1, (Params.n_followers-1)+2 + follower) = -1;
162  B(0,0)=1; B(1,0) = 1;
163  b(0) = Params.FollowerParam.capacities.at(follower);
