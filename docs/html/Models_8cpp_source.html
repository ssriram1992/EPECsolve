82 Models::EPEC::make_LL_QP(const LeadAllPar& Params, const unsigned int follower, Game::QP_Param* Foll, const Models::LeadLocs& Loc) const noexcept
83 {
84  const unsigned int LeadVars = Loc.at(Models::LeaderVars::End) - Params.n_followers;
85  arma::sp_mat Q(1,1), C(1, LeadVars + Params.n_followers - 1);
86  // Two constraints. One saying that you should be less than capacity
87  // Another saying that you should be less than leader imposed cap!
88  arma::sp_mat A(2, Loc.at(Models::LeaderVars::End) - 1), B(2, 1);
89  arma::vec c(1), b(2);
90  c.fill(0); b.fill(0);
91  A.zeros(); B.zeros(); C.zeros(); b.zeros(); Q.zeros(); c.zeros();
92  // Objective
93  Q(0, 0) = Params.FollowerParam.costs_quad.at(follower) + 2*Params.DemandParam.beta;
94  c(0) = Params.FollowerParam.costs_lin.at(follower) - Params.DemandParam.alpha;
95 
96  arma::mat Ctemp(1, Loc.at(Models::LeaderVars::End)-1, arma::fill::zeros);
97  Ctemp.cols(0, Params.n_followers-1).fill(Params.DemandParam.beta); // First n-1 entries and 1 more entry is Beta
98  Ctemp(0, Params.n_followers) = -Params.DemandParam.beta; // For q_exp
99 
100  Ctemp(0, (Params.n_followers-1)+2+Params.n_followers+follower ) = 1; // q_{-i}, then import, export, then tilde q_i, then i-th tax
101 
102  C = Ctemp;
103  A(1, (Params.n_followers-1)+2 + follower) = -1;
104  B(0,0)=1; B(1,0) = 1;
105  b(0) = Params.FollowerParam.capacities.at(follower);
106 
107  Foll->set(std::move(Q), std::move(C), std::move(A), std::move(B), std::move(c), std::move(b));
108 }

335 Models::EPEC::add_Leaders_tradebalance_constraints(const unsigned int i)
336 {
337  if (i>=this->nCountries) throw string("Error in Models::EPEC::add_Leaders_tradebalance_constraints. Bad argument");
338  int nImp = 0;
339  // Counts the number of countries from which the current country imports
340  for(auto val=TranspCosts.begin_col(i); val!=TranspCosts.end_col(i); ++val) nImp++;
341  // substitutes that answer to nImportMarkets at the current position
342  this->nImportMarkets.at(i) = (nImp);
343  // Adding the constraint that the sum of imports from all countries equals total imports
344  arma::vec a(nImp + this->countriesLL.at(i)->getNprimals(), arma::fill::zeros);
345  const auto n_followers = this->AllLeadPars.at(i).n_followers;

348  this->countriesLL.at(i)->addDummy(nImp).addLeadCons(a, 0).addLeadCons(-a,0);
349  Locations.at(i)[Models::LeaderVars::CountryImport] = Locations.at(i).at(Models::LeaderVars::End);
